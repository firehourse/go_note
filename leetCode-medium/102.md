Given the root of a binary tree, return the level order traversal of its nodes' values. (i.e., from left to right, level by level).

 

Example 1:


Input: root = [3,9,20,null,null,15,7]
Output: [[3],[9,20],[15,7]]
Example 2:

Input: root = [1]
Output: [[1]]


這題其實是一題基本的BFS
那廣度搜尋基本上你就是要遍歷N個節點，這裡的技巧就是用queue的方式來節省記憶體

```go
/**
 * Definition for a binary tree node.
 * type TreeNode struct {
 *     Val int
 *     Left *TreeNode
 *     Right *TreeNode
 * }
 */
func levelOrder(root *TreeNode) [][]int {
    if root == nil {
        return [][]int{}
    }
    result := make([][]int,0)
    q := make([]*TreeNode,0)
    q = append(q, root)    
    // queue 不為空即處理
    for len(q) != 0 {
        // 該層大小
        size := len(q)
        temp := make([]int, size)
        for i :=0;i<size;i++ {
            // 把值先塞入
            temp[i] =q[i].Val
            // 塞入左子樹
            if q[i].Left != nil {
                q = append(q, q[i].Left)
            }
            // 塞入右子樹
            if q[i].Right != nil {
                q = append(q, q[i].Right)
            }
        }
        result = append(result, temp)
        q = q[size:]
    }
    return result
}
